# 1235 Maximum Profit in Job Scheduling



```java
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        int len = startTime.length;
        int[][] time = new int[len][4];

        for(int i = 0; i < len; i++){
            time[i][0] = startTime[i];
            time[i][1] = endTime[i];
            time[i][2] = profit[i];
            time[i][3] = profit[i];
        }

        Arrays.sort(time, (o1, o2) -> (o1[1] - o2[1]));
        int maxProfit = 0;

        for(int i = 1; i < len; i++){
            time[i][3] = Math.max(time[i][2], time[i - 1][3]);
            for(int j = i  - 1; j >= 0; j--){
                if(time[j][1] <= time[i][0]){
                    time[i][3] = Math.max(time[i][3], time[j][3] + time[i][2]);
                    break;
                }
            }

            maxProfit = Math.max(maxProfit, time[i][3]);
        }

        return maxProfit;
    }
```

```java
    public int countServers(int[][] grid) {
        Set<String> set = new HashSet<>();
        int row = grid.length;
        int col = grid[0].length;
        for(int i = 0; i < row; i++){
            Set<String> temp = new HashSet<>();
            for(int j = 0; j < col; j++){
                if(grid[i][j] == 1)
                    temp.add(i + "@" + j);
            }

            if(temp.size() >= 2)
                set.addAll(temp);
        }

        for(int j = 0; j < col; j++){
            Set<String> temp = new HashSet<>();
            for(int i = 0; i < row; i++){
                if(grid[i][j] == 1)
                    temp.add(i + "@" + j);
            }

            if(temp.size() >= 2)
                set.addAll(temp);
        }

        return set.size();
    }
```

```java
class Solution {
    public int subtractProductAndSum(int n) {
        String str = n + "";

        int res1 = 1, res2 = 0;
        for(char ch : str.toCharArray()){
            res1 *= (ch - '0');
            res2 += (ch - '0');
        }

        return res1 - res2;
    }
}

func subtractProductAndSum(n int) int {
	res1 := 1
	res2 := 0
	
	for ;n != 0; {
		digit := n % 10
		
		res1 *= digit
		res2 += digit
		
		n /= 10
	}
	
	return res1 - res2
}
```
